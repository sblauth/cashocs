42def compute_boundary_distance( 

43 mesh: fenics.Mesh, 

44 boundaries: fenics.MeshFunction | None = None, 

45 boundary_idcs: list[int | str] | None = None, 

46 tol: float = 1e-1, 

47 max_iter: int = 10, 

48 minimum_distance: float = 0.0, 

49 method: str = "eikonal", 

50) -> fenics.Function: 

51 """Computes (an approximation of) the distance to the boundary. 

52 

53 The user can specify which boundaries are considered for the distance computation 

54 by specifying the parameters `boundaries` and `boundary_idcs`. Default is to 

55 consider all boundaries. 

56 

57 Args: 

58 mesh: The dolfin mesh object, representing the computational domain. 

59 boundaries: A meshfunction for the boundaries, which is needed in case specific 

60 boundaries are targeted for the distance computation (while others are 

61 ignored), default is `None` (all boundaries are used). 

62 boundary_idcs: A list of indices which indicate, which parts of the boundaries 

63 should be used for the distance computation, default is `None` (all 

64 boundaries are used). 

65 tol: A tolerance for the iterative solution of the eikonal equation. Default is 

66 1e-1. 

67 max_iter: Number of iterations for the iterative solution of the eikonal 

68 equation. Default is 10. 

69 minimum_distance: The distance of the mesh boundary to the (physical) wall. This 

70 should be set to 0.0 for most applications, which is also the default. One 

71 exception is for turbulence modeling and wall functions. 

72 method: Which method should be used to compute the boundary distance. Can either 

73 be 'poisson' or 'eikonal'. The default is 'eikonal', which is more accurate. 

74 The poisson approach is less accurate, but more robust and gives a very 

75 good approximation close to the wall. 

76 

77 Returns: 

78 A fenics function representing an approximation of the distance to the boundary. 

79 

80 """ 

81 if method == "poisson": 

82 return compute_boundary_distance_poisson( 

83 mesh, 

84 boundaries=boundaries, 

85 boundary_idcs=boundary_idcs, 

86 minimum_distance=minimum_distance, 

87 ) 

88 elif method == "eikonal": 

89 return compute_boundary_distance_eikonal( 

90 mesh, 

91 boundaries=boundaries, 

92 boundary_idcs=boundary_idcs, 

93 tol=tol, 

94 max_iter=max_iter, 

95 ) 

96 else: 

97 raise _exceptions.InputError( 

98 "compute_boundary_distance", 

99 "method", 

100 "The method can only be 'poisson' or 'eikonal'.", 

101 )

104def compute_boundary_distance_poisson( 

105 mesh: fenics.Mesh, 

106 boundaries: fenics.MeshFunction | None = None, 

107 boundary_idcs: list[int | str] | None = None, 

108 minimum_distance: float = 0.0, 

109) -> fenics.Function: 

110 """Computes the distance to the boundary with a Poisson approach. 

111 

112 The user can specify which boundaries are considered for the distance computation 

113 by specifying the parameters `boundaries` and `boundary_idcs`. Default is to 

114 consider all boundaries. 

115 

116 The approach is described, e.g., in Section 1.1.3 of 

117 `Tucker, Differential equation-based wall distance computation for DES and RANS 

118 (2003) <https://doi.org/10.1016/S0021-9991(03)00272-9>`_. 

119 

120 Args: 

121 mesh (fenics.Mesh): The dolfin mesh object, representing the computational 

122 domain 

123 boundaries (fenics.MeshFunction | None, optional): A meshfunction for the 

124 boundaries, which is needed in case specific boundaries are targeted for 

125 the distance computation (while others are ignored), default is `None` (all 

126 boundaries are used). Defaults to None. 

127 boundary_idcs (list[int | str] | None, optional): A list of indices which 

128 indicate, which parts of the boundaries should be used for the distance 

129 computation, default is `None` (all boundaries are used). Defaults to None. 

130 minimum_distance (float, optional): The distance of the mesh boundary to the 

131 (physical) wall. This should be set to 0.0 for most applications, which is 

132 also the default. One exception is for turbulence modeling and wall 

133 functions. Defaults to 0.0. 

134 

135 Returns: 

136 A fenics function representing an approximation of the distance to the boundary. 

137 

138 """ 

139 cg1_space = fenics.FunctionSpace(mesh, "CG", 1) 

140 dx = ufl.Measure("dx", domain=mesh) 

141 

142 u = fenics.Function(cg1_space) 

143 v = fenics.TestFunction(cg1_space) 

144 poisson_form = ( 

145 ufl.dot(ufl.grad(u), ufl.grad(v)) * dx - fenics.Constant(1.0) * v * dx 

146 ) 

147 

148 if (boundaries is not None) and (boundary_idcs is not None): 

149 if len(boundary_idcs) > 0: 

150 bcs = _utils.create_dirichlet_bcs( 

151 cg1_space, fenics.Constant(minimum_distance), boundaries, boundary_idcs 

152 ) 

153 else: 

154 bcs = fenics.DirichletBC( 

155 cg1_space, 

156 fenics.Constant(minimum_distance), 

157 fenics.CompiledSubDomain("on_boundary"), 

158 ) 

159 else: 

160 bcs = fenics.DirichletBC( 

161 cg1_space, 

162 fenics.Constant(minimum_distance), 

163 fenics.CompiledSubDomain("on_boundary"), 

164 ) 

165 

166 ksp_options: _typing.KspOption = { 

167 "ksp_type": "cg", 

168 "pc_type": "hypre", 

169 "pc_hypre_type": "boomeramg", 

170 "pc_hypre_boomeramg_strong_threshold": 0.7, 

171 "ksp_rtol": 1e-12, 

172 "ksp_atol": 1e-50, 

173 "ksp_max_it": 1000, 

174 } 

175 nonlinear_solvers.linear_solve(poisson_form, u, bcs, ksp_options=ksp_options) 

176 

177 distance = fenics.Function(cg1_space) 

178 form = ( 

179 distance * v * dx 

180 - ufl.sqrt(ufl.dot(ufl.grad(u), ufl.grad(u)) + fenics.Constant(2.0) * u) 

181 * v 

182 * dx 

183 + ufl.sqrt(ufl.dot(ufl.grad(u), ufl.grad(u))) * v * dx 

184 ) 

185 nonlinear_solvers.linear_solve(form, distance, bcs, ksp_options=ksp_options) 

186 

187 return distance

190def compute_boundary_distance_eikonal( 

191 mesh: fenics.Mesh, 

192 boundaries: fenics.MeshFunction | None = None, 

193 boundary_idcs: list[int | str] | None = None, 

194 tol: float = 1e-1, 

195 max_iter: int = 10, 

196) -> fenics.Function: 

197 """Computes the distance to the boundary by solving the Eikonal equation. 

198 

199 The function iteratively solves the Eikonal equation to compute the distance to the 

200 boundary. 

201 

202 The user can specify which boundaries are considered for the distance computation 

203 by specifying the parameters `boundaries` and `boundary_idcs`. Default is to 

204 consider all boundaries. 

205 

206 Args: 

207 mesh: The dolfin mesh object, representing the computational domain 

208 boundaries: A meshfunction for the boundaries, which is needed in case specific 

209 boundaries are targeted for the distance computation (while others are 

210 ignored), default is `None` (all boundaries are used). 

211 boundary_idcs: A list of indices which indicate, which parts of the boundaries 

212 should be used for the distance computation, default is `None` (all 

213 boundaries are used). 

214 tol: A tolerance for the iterative solution of the eikonal equation. Default is 

215 1e-1. 

216 max_iter: Number of iterations for the iterative solution of the eikonal 

217 equation. Default is 10. 

218 

219 Returns: 

220 A fenics function representing an approximation of the distance to the boundary. 

221 

222 """ 

223 function_space = fenics.FunctionSpace(mesh, "CG", 1) 

224 dx = measure.NamedMeasure("dx", mesh) 

225 

226 ksp_options = copy.deepcopy(_utils.linalg.iterative_ksp_options) 

227 

228 u = fenics.TrialFunction(function_space) 

229 v = fenics.TestFunction(function_space) 

230 

231 u_curr = fenics.Function(function_space) 

232 u_prev = fenics.Function(function_space) 

233 norm_u_prev = ufl.sqrt(ufl.dot(ufl.grad(u_prev), ufl.grad(u_prev))) 

234 

235 if (boundaries is not None) and (boundary_idcs is not None): 

236 if len(boundary_idcs) > 0: 

237 bcs = _utils.create_dirichlet_bcs( 

238 function_space, fenics.Constant(0.0), boundaries, boundary_idcs 

239 ) 

240 else: 

241 bcs = fenics.DirichletBC( 

242 function_space, 

243 fenics.Constant(0.0), 

244 fenics.CompiledSubDomain("on_boundary"), 

245 ) 

246 else: 

247 bcs = fenics.DirichletBC( 

248 function_space, 

249 fenics.Constant(0.0), 

250 fenics.CompiledSubDomain("on_boundary"), 

251 ) 

252 

253 lhs = ufl.dot(ufl.grad(u), ufl.grad(v)) * dx 

254 rhs = fenics.Constant(1.0) * v * dx 

255 

256 _utils.assemble_and_solve_linear(lhs, rhs, u_curr, bcs=bcs, ksp_options=ksp_options) 

257 

258 rhs = ufl.dot(ufl.grad(u_prev) / norm_u_prev, ufl.grad(v)) * dx 

259 

260 residual_form = ( 

261 pow( 

262 ufl.sqrt(ufl.dot(ufl.grad(u_curr), ufl.grad(u_curr))) 

263 - fenics.Constant(1.0), 

264 2, 

265 ) 

266 * dx 

267 ) 

268 

269 res_0 = np.sqrt(fenics.assemble(residual_form)) 

270 

271 for _ in range(max_iter): 

272 u_prev.vector().vec().aypx(0.0, u_curr.vector().vec()) 

273 u_prev.vector().apply("") 

274 _utils.assemble_and_solve_linear( 

275 lhs, rhs, u_curr, bcs=bcs, ksp_options=ksp_options 

276 ) 

277 res = np.sqrt(fenics.assemble(residual_form)) 

278 

279 if res <= res_0 * tol: 

280 break 

281 

282 return u_curr
