18"""Gradient problem. 

19 

20This class is used to solve the Riesz projection to obtain the gradient of the reduced 

21cost functional. 

22""" 

23 

24from __future__ import annotations 

25 

26import copy 

27from typing import TYPE_CHECKING 

28 

29import fenics 

30 

31from cashocs import _forms 

32from cashocs import _utils 

33from cashocs import log 

34from cashocs._pde_problems import pde_problem 

35 

36if TYPE_CHECKING: 

37 from cashocs import _typing 

38 from cashocs._database import database 

39 from cashocs._pde_problems import adjoint_problem as ap 

40 from cashocs._pde_problems import state_problem as sp 

41 

42 

43class ControlGradientProblem(pde_problem.PDEProblem): 

44 """A class representing the Riesz problem to determine the gradient.""" 

45 

46 def __init__( 

47 self, 

48 db: database.Database, 

49 form_handler: _forms.ControlFormHandler, 

50 state_problem: sp.StateProblem, 

51 adjoint_problem: ap.AdjointProblem, 

52 ) -> None: 

53 """Initializes self. 

54 

55 Args: 

56 db: The database of the problem. 

57 form_handler: The FormHandler object of the optimization problem. 

58 state_problem: The StateProblem object used to solve the state equations. 

59 adjoint_problem: The AdjointProblem used to solve the adjoint equations. 

60 

61 """ 

62 super().__init__(db) 

63 

64 self.form_handler = form_handler 

65 self.state_problem = state_problem 

66 self.adjoint_problem = adjoint_problem 

67 

68 self.gradient_norm_squared = 1.0 

69 

70 gradient_tol: float = self.config.getfloat( 

71 "OptimizationRoutine", "gradient_tol" 

72 ) 

73 

74 gradient_method: str = self.config.get("OptimizationRoutine", "gradient_method") 

75 

76 if db.parameter_db.gradient_ksp_options is not None: 

77 self.riesz_ksp_options = db.parameter_db.gradient_ksp_options 

78 else: 

79 if gradient_method.casefold() == "direct": 

80 option: _typing.KspOption = copy.deepcopy( 

81 _utils.linalg.direct_ksp_options 

82 ) 

83 elif gradient_method.casefold() == "iterative": 

84 option = copy.deepcopy(_utils.linalg.iterative_ksp_options) 

85 option["ksp_rtol"] = gradient_tol 

86 

87 self.riesz_ksp_options = [] 

88 for _ in range(len(self.db.function_db.gradient)): 

89 self.riesz_ksp_options.append(option) 

90 

91 self.b_tensors = [ 

92 fenics.PETScVector(self.db.geometry_db.mpi_comm) 

93 for _ in range(len(self.db.function_db.gradient)) 

94 ] 

95 

96 def solve(self) -> list[fenics.Function]: 

97 """Solves the Riesz projection problem to obtain the gradient. 

98 

99 Returns: 

100 The list containing the (components of the) gradient of the cost functional. 

101 

102 """ 

103 self.state_problem.solve() 

104 self.adjoint_problem.solve() 

105 

106 if not self.has_solution: 

107 log.begin("Computing the gradient.", level=log.DEBUG) 

108 for i in range(len(self.db.function_db.gradient)): 

109 self.form_handler.assemblers[i].assemble(self.b_tensors[i]) 

110 self.linear_solver.solve( 

111 self.db.function_db.gradient[i], 

112 A=self.form_handler.riesz_projection_matrices[i], 

113 b=self.b_tensors[i].vec(), 

114 ksp_options=self.riesz_ksp_options[i], 

115 ) 

116 

117 self.has_solution = True 

118 

119 self.gradient_norm_squared = self.form_handler.scalar_product( 

120 self.db.function_db.gradient, self.db.function_db.gradient 

121 ) 

122 

123 self.db.callback.call_post() 

124 log.end() 

125 

126 return self.db.function_db.gradient
