50class OptimalControlProblem(optimization_problem.OptimizationProblem): 

51 """Implements an optimal control problem. 

52 

53 This class is used to define an optimal control problem, and also to solve 

54 it subsequently. For a detailed documentation, see the examples in the 

55 :ref:`tutorial <tutorial_index>`. For easier input, when considering single 

56 (state or control) variables, these do not have to be wrapped into a list. 

57 Note, that in the case of multiple variables these have to be grouped into 

58 ordered lists, where state_forms, bcs_list, states, adjoints have to have 

59 the same order (i.e. ``[y1, y2]`` and ``[p1, p2]``, where ``p1`` is the adjoint of 

60 ``y1`` and so on). 

61 """

100 

101 Args: 

102 state_forms: The weak form of the state equation (user implemented). Can be 

103 either a single UFL form, or a (ordered) list of UFL forms. 

104 bcs_list: The list of :py:class:`fenics.DirichletBC` objects describing

112 :py:class:`fenics.Function`, or a list of these. 

113 adjoints: The adjoint variable(s), can either be a 

114 :py:class:`fenics.Function`, or a (ordered) list of these.

116 :py:func:`cashocs.load_config`. Alternatively, this can also be 

117 ``None``, in which case the default configurations are used, except for 

118 the optimization algorithm. This has then to be specified in the 

119 :py:meth:`solve <cashocs.OptimalControlProblem.solve>` method. The 

120 default is ``None``. 

121 riesz_scalar_products: The scalar products of the control space. Can either 

122 be None, a single UFL form, or a (ordered) list of UFL forms. If 

123 ``None``, the :math:`L^2(\Omega)` product is used (default is ``None``). 

124 control_constraints: Box constraints posed on the control, ``None`` means 

125 that there are none (default is ``None``). The (inner) lists should 

126 contain two elements of the form ``[u_a, u_b]``, where ``u_a`` is the 

127 lower, and ``u_b`` the upper bound. 

128 initial_guess: list of functions that act as initial guess for the state 

129 variables, should be valid input for :py:func:`fenics.assign`. Defaults

136 the same options as for the state systems are used (default is 

137 ``None``).

140 either a direct or an iterative method is used (depending on the 

141 configuration, section OptimizationRoutine, key gradient_method). 

142 desired_weights: A list of values for scaling the cost functional terms. If 

143 this is supplied, the cost functional has to be given as list of

146 that `desired_weights` is `None`, no scaling is performed. Default is 

147 `None`.

150 preconditioner_forms: The list of forms for the preconditioner. The default

154 solve of the state system 

155 post_callback: A function (without arguments) that will be called after the 

156 computation of the gradient. 

157 linear_solver: The linear solver (KSP) which is used to solve the linear 

158 systems arising from the discretized PDE. 

159 adjoint_linear_solver: The linear solver (KSP) which is used to solve the 

160 (linear) adjoint system. 

161 newton_linearizations: A (list of) UFL forms describing which (alternative) 

162 linearizations should be used for the (nonlinear) state equations when 

163 solving them (with Newton's method). The default is `None`, so that the 

164 Jacobian of the supplied state forms is used. 

165 excluded_from_time_derivative: For each state equation, a list of indices 

166 which are not part of the first order time derivative for pseudo time 

167 stepping. Example: Pressure for incompressible flow. Default is None. 

168 

169 Examples:

312 def _setup_solver(self) -> optimization_algorithms.OptimizationAlgorithm: 

313 line_search_type = self.config.get("LineSearch", "method").casefold() 

314 if line_search_type == "armijo": 

315 line_search: ls.LineSearch = ls.ArmijoLineSearch(self.db, self) 

316 elif line_search_type == "polynomial": 

317 line_search = ls.PolynomialLineSearch(self.db, self) 

318 else: 

319 raise _exceptions.CashocsException("This code cannot be reached.") 

320 

321 if self.algorithm.casefold() == "newton": 

322 self.form_handler.hessian_form_handler.compute_newton_forms() 

323 self.hessian_problem = _pde_problems.HessianProblem( 

324 self.db, 

325 self.form_handler, 

326 self.general_form_handler.adjoint_form_handler, 

327 self.gradient_problem, 

328 self.box_constraints, 

329 ) 

330 

331 if self.algorithm.casefold() == "gradient_descent": 

332 solver: optimization_algorithms.OptimizationAlgorithm = ( 

333 optimization_algorithms.GradientDescentMethod( 

334 self.db, self, line_search 

335 ) 

336 ) 

337 elif self.algorithm.casefold() == "lbfgs": 

338 solver = optimization_algorithms.LBFGSMethod(self.db, self, line_search) 

339 elif self.algorithm.casefold() == "conjugate_gradient": 

340 solver = optimization_algorithms.NonlinearCGMethod( 

341 self.db, self, line_search 

342 ) 

343 elif self.algorithm.casefold() == "newton": 

344 solver = optimization_algorithms.NewtonMethod(self.db, self, line_search) 

345 elif self.algorithm.casefold() == "none": 

346 raise _exceptions.InputError( 

347 "cashocs.OptimalControlProblem.solve", 

348 "algorithm", 

349 "You did not specify a solution algorithm in your config file. " 

350 "You have to specify one in the solve method. Needs to be one of" 

351 "'gradient_descent' ('gd'), 'lbfgs' ('bfgs'), 'conjugate_gradient' " 

352 "('cg'), or 'newton'.", 

353 ) 

354 else: 

355 raise _exceptions.CashocsException("This code cannot be reached.") 

356 

357 return solver

359 def solve( 

360 self, 

361 algorithm: str | None = None, 

362 rtol: float | None = None, 

363 atol: float | None = None, 

364 max_iter: int | None = None, 

365 ) -> None: 

366 r"""Solves the optimization problem by the method specified in the config file. 

367 

368 Updates / overwrites states, controls, and adjoints according 

369 to the optimization method, i.e., the user-input :py:func:`fenics.Function` 

370 objects. 

371 

372 Args: 

373 algorithm: Selects the optimization algorithm. Valid choices are 

374 ``'gradient_descent'`` or ``'gd'`` for a gradient descent method, 

375 ``'conjugate_gradient'``, ``'nonlinear_cg'``, ``'ncg'`` or ``'cg'`` 

376 for nonlinear conjugate gradient methods, ``'lbfgs'`` or ``'bfgs'`` for 

377 limited memory BFGS methods, and ``'newton'`` for a truncated Newton 

378 method. This overwrites the value specified in the config file. If this 

379 is ``None``, then the value in the config file is used. Default is 

380 ``None``. 

381 rtol: The relative tolerance used for the termination criterion. Overwrites 

382 the value specified in the config file. If this is ``None``, the value 

383 from the config file is taken. Default is ``None``. 

384 atol: The absolute tolerance used for the termination criterion. Overwrites 

385 the value specified in the config file. If this is ``None``, the value 

386 from the config file is taken. Default is ``None``. 

387 max_iter: The maximum number of iterations the optimization algorithm can 

388 carry out before it is terminated. Overwrites the value specified in the 

389 config file. If this is ``None``, the value from the config file is 

390 taken. Default is ``None``. 

391 

392 Notes: 

393 If either ``rtol`` or ``atol`` are specified as arguments to the ``.solve`` 

394 call, the termination criterion changes to: 

395 - a purely relative one (if only ``rtol`` is specified), i.e., 

396 

397 .. math:: 

398 || \nabla J(u_k) || \leq \texttt{rtol} || \nabla J(u_0) ||. 

399 

400 - a purely absolute one (if only ``atol`` is specified), i.e., 

401 

402 .. math:: 

403 || \nabla J(u_K) || \leq \texttt{atol}. 

404 

405 - a combined one if both ``rtol`` and ``atol`` are specified, i.e., 

406 

407 .. math:: 

408 || \nabla J(u_k) || \leq \texttt{atol} 

409 + \texttt{rtol} || \nabla J(u_0) || 

410 

411 """ 

412 if not self._silent: 

413 log.begin("Solving the optimal control problem.", level=log.INFO) 

414 super().solve(algorithm=algorithm, rtol=rtol, atol=atol, max_iter=max_iter) 

415 

416 self._setup_control_bcs() 

417 

418 self.solver = self._setup_solver() 

419 self.solver.run() 

420 self.solver.post_processing() 

421 if not self._silent: 

422 log.end()
