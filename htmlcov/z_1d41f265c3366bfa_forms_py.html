38def summation(x: list[T]) -> T | fenics.Constant: 

39 """Sums elements of a list in a UFL friendly fashion. 

40 

41 This can be used to sum, e.g., UFL forms, or UFL expressions that can be used in UFL 

42 forms. 

43 

44 Args: 

45 x: The list of entries that shall be summed. 

46 

47 Returns: 

48 Sum of input (same type as entries of input). 

49 

50 Notes: 

51 For "usual" summation of integers or floats, the built-in sum function 

52 of python or the numpy variant are recommended. Still, they are 

53 incompatible with FEniCS objects, so this function should be used for 

54 the latter. 

55 

56 """ 

57 if len(x) == 0: 

58 y = fenics.Constant(0.0) 

59 log.warning("Empty list handed to summation, returning 0.") 

60 else: 

61 y = x[0] 

62 

63 for item in x[1:]: 

64 y += item # type: ignore 

65 

66 return y

69def multiplication(x: list[T]) -> T | fenics.Constant: 

70 """Multiplies the elements of a list in a UFL friendly fashion. 

71 

72 Used to build the product of certain UFL expressions to construct a UFL form. 

73 

74 Args: 

75 x: The list whose entries shall be multiplied. 

76 

77 Returns: 

78 The result of the multiplication. 

79 

80 """ 

81 if len(x) == 0: 

82 y = fenics.Constant(1.0) 

83 log.warning("Empty list handed to multiplication, returning 1.") 

84 else: 

85 y = x[0] 

86 

87 for item in x[1:]: 

88 y *= item # type: ignore 

89 

90 return y

93def max_(a: float | fenics.Function, b: float | fenics.Function) -> ufl.core.expr.Expr: 

94 """Computes the maximum of a and b. 

95 

96 Args: 

97 a: The first parameter. 

98 b: The second parameter. 

99 

100 Returns: 

101 The maximum of a and b. 

102 

103 """ 

104 return (a + b + abs(a - b)) / fenics.Constant(2.0)

121def moreau_yosida_regularization( 

122 term: ufl.core.expr.Expr, 

123 gamma: float, 

124 measure: ufl.Measure, 

125 lower_threshold: float | fenics.Function | None = None, 

126 upper_threshold: float | fenics.Function | None = None, 

127 shift_lower: float | fenics.Function | None = None, 

128 shift_upper: float | fenics.Function | None = None, 

129) -> ufl.Form: 

130 r"""Implements a Moreau-Yosida regularization of an inequality constraint. 

131 

132 The general form of the inequality is of the form :: 

133 

134 lower_threshold <= term <= upper_threshold 

135 

136 which is defined over the region specified in ``measure``. 

137 

138 In case ``lower_threshold`` or ``upper_threshold`` are ``None``, they are set to 

139 :math:`-\infty` and :math:`\infty`, respectively. 

140 

141 Args: 

142 term: The term inside the inequality constraint. 

143 gamma: The weighting factor of the regularization. 

144 measure: The measure over which the inequality constraint is defined. 

145 lower_threshold: The lower threshold for the inequality constraint. In case this 

146 is ``None``, the lower bound is set to :math:`-\infty`. The default is 

147 ``None`` 

148 upper_threshold: The upper threshold for the inequality constraint. In case this 

149 is ``None``, the upper bound is set to :math:`\infty`. The default is 

150 ``None`` 

151 shift_lower: A shift function for the lower bound of the Moreau-Yosida 

152 regularization. Should be non-positive. In case this is ``None``, it is set 

153 to 0. Default is ``None``. 

154 shift_upper: A shift function for the upper bound of the Moreau-Yosida 

155 regularization. Should be non-negative. In case this is ``None``, it is set 

156 to 0. Default is ``None``. 

157 

158 Returns: 

159 The ufl form of the Moreau-Yosida regularization, to be used in the cost 

160 functional. 

161 

162 """ 

163 reg_lower = None 

164 reg_upper = None 

165 

166 if lower_threshold is None and upper_threshold is None: 

167 raise _exceptions.InputError( 

168 "cashocs._utils.moreau_yosida_regularization", 

169 "upper_threshold, lower_threshold", 

170 "At least one of the threshold parameters has to be defined.", 

171 ) 

172 

173 if shift_lower is None: 

174 shift_lower = fenics.Constant(0.0) 

175 if shift_upper is None: 

176 shift_upper = fenics.Constant(0.0) 

177 

178 reg = [] 

179 

180 if lower_threshold is not None: 

181 reg_lower = ( 

182 fenics.Constant(1 / (2 * gamma)) 

183 * pow( 

184 min_( 

185 shift_lower + fenics.Constant(gamma) * (term - lower_threshold), 

186 fenics.Constant(0.0), 

187 ), 

188 2, 

189 ) 

190 * measure 

191 ) 

192 reg.append(reg_lower) 

193 if upper_threshold is not None: 

194 reg_upper = ( 

195 fenics.Constant(1 / (2 * gamma)) 

196 * pow( 

197 max_( 

198 shift_upper + fenics.Constant(gamma) * (term - upper_threshold), 

199 fenics.Constant(0.0), 

200 ), 

201 2, 

202 ) 

203 * measure 

204 ) 

205 reg.append(reg_upper) 

206 

207 return summation(reg)

210def create_dirichlet_bcs( 

211 function_space: fenics.FunctionSpace, 

212 value: fenics.Constant | fenics.Expression | fenics.Function | float | tuple[float], 

213 boundaries: fenics.MeshFunction, 

214 idcs: list[int | str] | int | str, 

215 **kwargs: Any, 

216) -> list[fenics.DirichletBC]: 

217 """Create several Dirichlet boundary conditions at once. 

218 

219 Wraps multiple Dirichlet boundary conditions into a list, in case 

220 they have the same value but are to be defined for multiple boundaries 

221 with different markers. Particularly useful for defining homogeneous 

222 boundary conditions. 

223 

224 Args: 

225 function_space: The function space onto which the BCs should be imposed on. 

226 value: The value of the boundary condition. Has to be compatible with the 

227 function_space, so that it could also be used as 

228 ``fenics.DirichletBC(function_space, value, ...)``. 

229 boundaries: The :py:class:`fenics.MeshFunction` object representing the 

230 boundaries. 

231 idcs: A list of indices / boundary markers that determine the boundaries 

232 onto which the Dirichlet boundary conditions should be applied to. 

233 Can also be a single entry for a single boundary. If your mesh file 

234 is named, then you can also use the names of the boundaries to define the 

235 boundary conditions. 

236 **kwargs: Keyword arguments for fenics.DirichletBC 

237 

238 Returns: 

239 A list of DirichletBC objects that represent the boundary conditions. 

240 

241 Examples: 

242 Generate homogeneous Dirichlet boundary conditions for all 4 sides of 

243 the unit square :: 

244 

245 import fenics 

246 import cashocs 

247 

248 mesh, _, _, _, _, _ = cashocs.regular_mesh(25) 

249 V = fenics.FunctionSpace(mesh, 'CG', 1) 

250 bcs = cashocs.create_dirichlet_bcs(V, fenics.Constant(0), boundaries, 

251 [1,2,3,4]) 

252 

253 """ 

254 mesh = function_space.mesh() 

255 

256 if not isinstance(idcs, list): 

257 idcs = [idcs] 

258 

259 bcs_list = [] 

260 for entry in idcs: 

261 int_tag = _utils.tag_to_int(mesh, entry, "ds") 

262 bcs_list.append( 

263 fenics.DirichletBC(function_space, value, boundaries, int_tag, **kwargs) 

264 ) 

265 

266 return bcs_list

269def bilinear_boundary_form_modification(forms: list[ufl.Form]) -> list[ufl.Form]: 

270 """Modifies a bilinear form for the case it is given on the boundary only. 

271 

272 This avoids a bug in fenics.SystemAssembler where the matrices' sparsity pattern 

273 is not initialized correctly. 

274 

275 """ 

276 mod_forms = [] 

277 for form in forms: 

278 trial, test = form.arguments() 

279 mesh = trial.function_space().mesh() 

280 dx = ufl.Measure("dx", domain=mesh) 

281 mod_forms.append(form + fenics.Constant(0.0) * ufl.dot(trial, test) * dx) 

282 

283 return mod_forms
